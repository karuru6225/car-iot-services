  auto imu_update = AtomS3.Imu.update();
  if (imu_update) {
      auto data = AtomS3.Imu.getImuData();

      // The data obtained by getImuData can be used as follows.
      data.accel.x;      // accel x-axis value.
      data.accel.y;      // accel y-axis value.
      data.accel.z;      // accel z-axis value.
      data.accel.value;  // accel 3values array [0]=x / [1]=y / [2]=z.

      data.gyro.x;      // gyro x-axis value.
      data.gyro.y;      // gyro y-axis value.
      data.gyro.z;      // gyro z-axis value.
      data.gyro.value;  // gyro 3values array [0]=x / [1]=y / [2]=z.

      data.value;  // all sensor 9values array [0~2]=accel / [3~5]=gyro /

      AtomS3.Display.setCursor(0, 0);
      AtomS3.Display.setTextSize(2)
      AtomS3.Display.printf("ax:%f  ay:%f  az:%f\r\n", data.accel.x, data.accel.y,
                    data.accel.z);
  }
